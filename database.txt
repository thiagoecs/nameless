CREATE TABLE `users` (
  `id` int PRIMARY KEY NOT NULL AUTO_INCREMENT,
  `nickname` varchar(255),
  `password` varchar(255),
  `email` varchar(255),
  `userType` int,
  `avatarUrl` text
);

CREATE TABLE `posts` (
  `id` int PRIMARY KEY NOT NULL AUTO_INCREMENT,
  `restaurant` varchar(255),
  `description` varchar(255),
  `views` int DEFAULT '0',
  `votes` int DEFAULT '0',
  `creator` int,
  `createdAt` timestamp DEFAULT CURRENT_TIMESTAMP,
  `foodType` int
);

CREATE TABLE `files` (
  `postId` int,
  `sourceFile` text
);
